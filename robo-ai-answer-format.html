ROBO.AI : Answer Format

Public Answer Format

These are the types of answers that the ROBO.AI API will render.

Message Type Description Syntax Example
text To display a simple text answer 
{
"type":"text", 
"text": <text>
}
 
{
"type":"text", 
"text": "example text message"
}
html To display an html text answer 
{
"type":"html", 
"text": <html>
}
 
{
"type":"html", 
"text": "<p>a paragraph</p>"
}
cms To display a message from the CMS 
{
"type":"cms", 
"cmsId" : <id>,
"parameters": ["param1", "param2"]
}
 
{
"type":"cms", 

"cmsId" : "cms-answerid",

"parameters": ["param1", "param2"]
}
multichoice To show options to the user (buttons or chips) 
{
"type":"multichoice", 
"options": [ 
  {"label":<label>,"value":<value>},
  {"label":<label>,"value":<value>}
]
}
 
{
  "type":"multichoice", 
  "options": [ 
      {"label":"ja","value":"ja"},
      {"label":"nein","value":"nein"}
   ]
}
hints To show topic hints 
{
"type":"hints", 
"options": [ 
  {"label":<label>,"value":<value>},
  {"label":<label>,"value":<value>}
]
}
 
{
  "type":"hints", 
  "options": [ 
      {"label":"EQ Ready Check","value":"EQ Ready Check"},
      {"label":"EQ Reichweite","value":"EQ Reichweite"}
   ]
}
image To embed an image 
{
"type":"image", 
"title": <title>,
"thumbnailURL": <thumbnail-filename>, 
"url":<image-filename>
}
 
{
  "type":"image", 
  "title": "cool picture title",
  "thumbnailURL": "thumb.jpg", 
  "url":"picture.jpg"
}
 
carousel
 To embed a carousel 
{
"type":"carousel", 
"options": [
{"title": <title>,
"thumbnailURL": <thumbnail-filename>, 
"url":<image-filename>}]
}
 
{
  "type":"carousel", 
  "options": [

  {"title": "smart fortwo",
  "thumbnailURL": "fortwo.jpg", 
  "url":"fortwo.jpg"},
  {"title": "smart forfour",
  "thumbnailURL": "forfour.jpg", 
  "url":"forfour.jpg"}]
}
video To embed an attached video 
{
  "type":"video", 
  "title": <title>,
  "thumbnailURL": <thumbnail-filename>, 
  "url":<video-filename>
}
 
{
  "type":"video", 
  "title": "cool video title",
  "thumbnailURL": "video-thumb.jpg",      
  "url":"video.mp4"
}
youtube To embed a youtube video 
{
  "type":"youtube", 
  "title": <title>,
  "thumbnailURL": <thumbnail-filename>,
  "youtubeId": <youtube-id>
}
 
{
  "type":"youtube",
  "title": "cool video tittle",
  "thumbnailURL": "youtube-video.jpg", 
  "youtubeId":"A8-hshCnNGU"
}
links To show links 
{
"type":"links", 
"links": [ 
  {"title":<label>,"url":<value>},
  {"title":<label>,"url":<value>}
]
}
 
{
"type":"links", 
"links": [ 
  {"title":"link title","url":"http://url.com"},
  {"title":"link title","url":"http://url.com"}
]
}
command To send a command to the client 
{
"type":"command", 
"name": <title>,
"params": [ 
  {"key":<key>,"value":<value>},
  {"key":<key>,"value":<value>}
]
}
 
{
"type":"command", 
"name": "open-url",
"params": [ 
  {"key":"url","value":"http://google.com"}
]
}
ssml An answer in SSML format for voice bots. 
{
"type":"ssml", 
"ssml": <ssml>
}
{ "type": "ssml", "ssml": "<speak> Here is a number read as a cardinal number: <say-as interpret-as='cardinal'>12345</say-as>. Here is a word spelled out: <say-as interpret-as='spell-out'>hello</say-as>. </speak>" }
